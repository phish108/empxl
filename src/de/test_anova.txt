ANOVA.TEST = LAMBDA(
    dep_var; indep_var; [headers];
    LET(
        n_j; GROUP.COUNT(indep_var);
        group_k; ANZAHL(n_j);
        N_all; ANZAHL(dep_var);
        df_1; group_k - 1;
        df_2; N_all - group_k;
        mu_all; MITTELWERT(dep_var);
        mu_j; GROUP.MEAN(dep_var; indep_var);
        m_groups; 1*(indep_var = MTRANS(GROUP.LABELS(indep_var)));
        t_SSB; SUMME(n_j * (mu_j - mu_all)^2);
        t_SSE; SUMME((m_groups * dep_var - m_groups * MTRANS(mu_j))^2);
        F_value; (t_SSB / df_1) / (t_SSE / df_2);
        p_value; F.VERT.RE(F_value; df_1; df_2);
        results; HSTAPELN(F_value; df_1; df_2; p_value);
        wenn(oder(WURDEAUSGELASSEN(headers); nicht(headers));
            results;
            VSTAPELN(
                HSTAPELN("F"; "df_1"; "df_2"; "p");
                results
            )
        )
    )
);;
