T.TEST.S = LAMBDA( 
    dep_var; indep_var; [alternative]; [paired]; 
    LET(
        ab_names; EINDEUTIG(indep_var);
        paired; WENN(WURDEAUSGELASSEN(paired); 3; wenn(paired; 1; 3));
        alternative; WENN(WURDEAUSGELASSEN(alternative); "t"; PREP.ALTERNATIVE(alternative));
        altV; 1 + (alternative = "t");
        altO; alternative = "g";
        g_1; ZEILENWAHL(ab_names; 1);
        g_2; ZEILENWAHL(ab_names; 2);
        p_v; WENN(
            ANZAHL2(ab_names) = 2; 
            T.TEST(FILTER(dep_var; indep_var = g_1); FILTER(dep_var; indep_var = g_2); altV; paired);
            #WERT!
        );
        mw_1; MITTELWERT(FILTER(dep_var; indep_var = g_1));
        mw_2; MITTELWERT(FILTER(dep_var; indep_var = g_2));
        
        // The t-Test function already adjusts the p-value to two sided hypotheses.
        HSTAPELN(
            WENN(ODER(UND(altO; mw_1 < mw_2); UND(alternative = "l"; mw_1 > mw_2)); 1 - p_v; p_v);
            TEXTVERKETTEN(
                            ersterwert(
                                alternative; 
                                "t"; "<>"; 
                                "l"; "<"; 
                                "g"; ">"
                            );;
                            ab_names
                        )
        )
    )
);;
