
VREVERSE = LAMBDA(
    vector; 
    LET(
        v_length; VLENGTH(vector);
        v_pos; SEQUENZ(v_length;1;v_length;-1);
        WENN(
            SPALTEN(vector) = v_length;
            SPALTENWAHL(vector; v_pos);
            ZEILENWAHL(vector; v_pos)
        )
    )
);;

VEKTOR.UMKEHREN=LAMBDA(vektor; VREVERSE(vektor));;
