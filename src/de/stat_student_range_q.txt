# See https://en.wikipedia.org/wiki/Studentized_range_distribution
=LAMBDA(
    values; groups;
    LET(
        mean_i; GROUP.MEAN(values; groups);
        y_min; MIN(mean_i);
        y_max; MAX(mean_i);
        n_all; ANZAHL2(values);
        s_2; STAT.POOLED.VARIANCE(values; groups);
        (y_max - y_min) / SQRT(s_2 / (2 * n_all))
    )
)
