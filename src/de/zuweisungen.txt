// DESKTRIPTIVE STATISTIK
STAT.DESKRIPTIV=lambda(daten; [skalenniveau]; [namen]; [mit_üeberschrift]; 
    LET(
        type; WENN(wurdeausgelassen(skalenniveau); "m"; skalenniveau);
        labels; WENN(WURDEAUSGELASSEN(mit_überschrift); FALSCH; mit_überschrift);
        WENN(WURDEAUSGELASSEN(namen); 
            STAT.DESCRIBE(daten; type; ; mit_üeberschrift);
            STAT.DESCRIBE(daten; type; namen; mit_üeberschrift)
        )
    )
);;

STAT.DESKRIPTIV.MERKMALE=LAMBDA(daten; [variable]; WENN(WURDEAUSGELASSEN(variable); STAT.DESCRIBE.NOMINAL(daten);STAT.DESCRIBE.NOMINAL(daten;variable)));;

    variable_a; 
    [variable_b_oder_überschrift]; 
    [mit_überschrift]; 
    WENNS(
        UND(WURDEAUSGELASSEN(variable_b_oder_überschrift); WURDEAUSGELASSEN(mit_überschrift));
        STAT.TABLE(variable_a);
        WURDEAUSGELASSEN(variable_b_oder_überschrift);
        STAT.TABLE(variable_a;; mit_überschrift);

        STAT.TABLE(variable_a; variable_b_oder_überschrift; mit_überschrift)
    )
);;
