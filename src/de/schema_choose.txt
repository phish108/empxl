    [type];

        type; WENN(WURDEAUSGELASSEN(type); "m"; LINKS(KLEIN(type);1));
        types; LINKS(KLEIN(types);1);

            SPALTENWAHL(
                data;
                FILTER(
                    SEQUENZ(ANZAHL2(types));

                )
            )

    )
