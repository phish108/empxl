QQ.DATA = LAMBDA(
    values; [with_headers];
    LET(
        with_headers; ODER(WURDEAUSGELASSEN(with_headers); with_headers);  
        m_w; MITTELWERT(values);
        m_s; STABW.S(values);
        rank; (RANG.MITTELW(values; values; 1) - 0.5)/ANZAHL(values);
        results; HSTAPELN(
            NORM.S.INV(rank);
            (values - m_w) / m_s
        );
        WENN(
            with_headers;
            VSTAPELN(
                MTRANS({"Theoretisch"; "Gemessen"});
                results
            );
            results
        )
    )
);;
