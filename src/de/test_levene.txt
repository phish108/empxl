// Levene's test is a ANOVA on the reresiduals on the group  medians
// This implementation is the same as R's implementation at https://github.com/cran/car/blob/master/R/leveneTest.R

LEVENE.TEST = LAMBDA(
    x_i; y_i; [headers];
    LET(
        grps; SORTIEREN(EINDEUTIG(y_i));
        meds; GROUP.MEDIAN(x_i; y_i);
        resp; ABS(x_i - XVERWEIS(y_i; grps; meds));
        results; ANOVA.TEST(resp; y_i);
        wenn(oder(WURDEAUSGELASSEN(headers); nicht(headers));
            results;
            VSTAPELN(
                HSTAPELN("W"; "df_1"; "df_2"; "p");
                results
            )
        )
    )
);;
