=LAMBDA(
    p_size; [p_lower]; [p_strict];
    LET(
        t_lower; WENN(WURDEAUSGELASSEN(p_lower); TRUE; p_lower);
        t_strict; WENN(WURDEAUSGELASSEN(p_strict); FALSE; p_strict);
        WENN(
            UND(ANZAHL(p_size) = 1; ANZAHL2(t_lower) = 1; ANZAHL2(t_strict) = 1); 
            LET(
                vector; SEQUENZ(size); 
                tvector; MTRANS(vector); 
                1 * WENNS(
                    UND(lower; strict); vector > tvector; 
                    lower; vector >= tvector; 
                    strict; vector < tvector; 
                    WAHR; vector <= tvector
                )
            );
            #WERT!
        )
    ) 
)
