REGRESSION.RESIDUALS=LAMBDA(
    dep_var; indep_var;
    LET(
        KOEF; VREVERSE(MTRANS(RGP(dep_var;indep_var)));
        KOEFMAT; HSTAPELN(SEQUENZ(ZEILEN(dep_var);1;1;0); indep_var);
        ESTIMATES; MMULT(KOEFMAT; KOEF);
        ESTIMATES - dep_var
    )
);;

RGP.RESIDUALS=REGRESSION.RESIDUALS;;
